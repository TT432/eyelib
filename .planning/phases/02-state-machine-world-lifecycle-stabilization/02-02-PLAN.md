private static void handleStabilize() {
    Minecraft mc = Minecraft.getInstance();

    if (mc.level == null) {
        LOGGER.warn("[ClientSmoke] Level unexpectedly null during stabilization ‚Äî returning to WORLD_WAIT");
        state = ClientSmokeState.WORLD_WAIT;
        return;
    }

    long waitedTicks = mc.level.getGameTime() - stabilizeStartTick;
    long requiredTicks = ClientSmokeConfig.RELOAD_STABILIZE_TICKS.get();

    if (waitedTicks >= requiredTicks) {
        // Stabilization complete ‚Äî Phase 2 endpoint reached
        LOGGER.info("[ClientSmoke] Stabilization complete ‚Äî waited {} ticks (required: {}). World ready for test execution.",
                waitedTicks, requiredTicks);
        LOGGER.info("[ClientSmoke] Phase 2 complete ‚Äî {} test(s) queued. Phase 3 will execute tests.",
                discoveredTests.size());
        // STAY in STABILIZE state ‚Äî Phase 3 picks up from here
        // Do NOT advance to TEST_EXEC (that's Phase 3's responsibility)
    }
    // else: stay in STABILIZE ‚Äî no log spam, just wait
}
