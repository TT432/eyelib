// Extract text between "clientSmoke {" and the matching closing brace
private static String extractBlock(String source, String blockName) {
    int start = source.indexOf(blockName + " {");
    if (start < 0) return "";
    int braceCount = 0;
    int i = start + blockName.length() + 1; // after opening brace
    StringBuilder sb = new StringBuilder();
    for (; i < source.length(); i++) {
        char c = source.charAt(i);
        if (c == '{') braceCount++;
        else if (c == '}') {
            if (braceCount == 0) break;
            braceCount--;
        }
        sb.append(c);
    }
    return sb.toString();
}
